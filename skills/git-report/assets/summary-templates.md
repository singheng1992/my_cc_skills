# 报告总结模板
# 创建时间：2025-03-09
# 说明：定义不同报告类型的总结模板，支持变量插值
# 作者：Claude (MiniMax-M2.5)

# =================================================================
# 可用变量列表
# =================================================================
# {{date_label}}        - 日期范围标签
# {{project_count}}     - 有提交的项目数
# {{total_commits}}     - 总提交数
# {{start_date}}        - 开始日期
# {{end_date}}          - 结束日期
# {{generated_at}}      - 生成时间

# =================================================================
# 默认总结模板
# =================================================================
default: |
  在{{date_label}}期间，共涉及{{project_count}}个项目，完成{{total_commits}}次代码提交。
  工作内容持续进行中，具体详见上述项目提交记录。

# =================================================================
# 日报总结模板
# =================================================================
daily: |
  {{date_label}}的工作进展：
  - 涉及项目：{{project_count}}个
  - 提交次数：{{total_commits}}次
  - 工作按计划推进，详见各项目提交记录。

# =================================================================
# 周报总结模板
# =================================================================
weekly: |
  本周工作总结（{{date_label}}）：
  - 活跃项目：{{project_count}}个
  - 累计提交：{{total_commits}}次
  - 各项目开发工作有序推进，具体详见上述记录。

# =================================================================
# 月报总结模板
# =================================================================
monthly: |
  {{date_label}}工作综述：
  本月共在{{project_count}}个项目上进行开发，累计提交{{total_commits}}次。
  工作重点聚焦于项目功能迭代和优化，详见各项目详细记录。

# =================================================================
# 简洁模板
# =================================================================
simple: |
  {{date_label}}：{{project_count}}个项目，{{total_commits}}次提交。

# =================================================================
# 空模板（由AI生成）
# =================================================================
ai: |
  _AI_GENERATED_
